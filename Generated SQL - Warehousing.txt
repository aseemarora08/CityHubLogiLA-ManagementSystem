CREATE TABLE WAREHOUSE
(
  WarehouseID VARCHAR(50) NOT NULL,
  Address VARCHAR(100) NOT NULL,
  Email VARCHAR(50) NOT NULL,
  Phone INT NOT NULL,
  PRIMARY KEY (WarehouseID)
);

CREATE TABLE BIN
(
  BinID VARCHAR(50) NOT NULL,
  WarehouseID VARCHAR(50) NOT NULL,
  PRIMARY KEY (BinID),
  FOREIGN KEY (WarehouseID) REFERENCES WAREHOUSE(WarehouseID)
);

CREATE TABLE EMPLOYEE
(
  EmpID VARCHAR(50) NOT NULL,
  Name VARCHAR(50) NOT NULL,
  DOB DATE NOT NULL,
  Phone INT NOT NULL,
  Email VARCHAR(50) NOT NULL,
  Address VARCHAR(100) NOT NULL,
  WarehouseID VARCHAR(50) NOT NULL,
  PRIMARY KEY (EmpID),
  FOREIGN KEY (WarehouseID) REFERENCES WAREHOUSE(WarehouseID)
);

CREATE TABLE MANAGER
(
  ManagerID VARCHAR(50) NOT NULL,
  Name VARCHAR(50) NOT NULL,
  DOB DATE NOT NULL,
  Phone INT NOT NULL,
  Email VARCHAR(50) NOT NULL,
  Address VARCHAR(50) NOT NULL,
  WarehouseID VARCHAR(50) NOT NULL,
  PRIMARY KEY (ManagerID),
  FOREIGN KEY (WarehouseID) REFERENCES WAREHOUSE(WarehouseID)
);

CREATE TABLE PRODUCT_CATEGORY
(
  CategoryID VARCHAR(50) NOT NULL,
  Name VARCHAR(50) NOT NULL,
  Description VARCHAR(50) NOT NULL,
  PRIMARY KEY (CategoryID)
);

CREATE TABLE MNAUFACTURER
(
  ManufacturerID INT NOT NULL,
  Name VARCHAR(50) NOT NULL,
  Address VARCHAR( 100) NOT NULL,
  Phone INT NOT NULL,
  Address VARCHAR(50) NOT NULL,
  PRIMARY KEY (ManufacturerID)
);

CREATE TABLE CLIENT
(
  ClientID VARCHAR(50) NOT NULL,
  Name VARCHAR(50) NOT NULL,
  Phone INT NOT NULL,
  Address VARCHAR(100) NOT NULL,
  Email VARCHAR(50) NOT NULL,
  PRIMARY KEY (ClientID)
);

CREATE TABLE ORDER
(
  OrderID VARCHAR(50) NOT NULL,
  OrderDate DATE NOT NULL,
  ClientID VARCHAR(50) NOT NULL,
  PRIMARY KEY (OrderID),
  FOREIGN KEY (ClientID) REFERENCES CLIENT(ClientID)
);

CREATE TABLE PRODUCT
(
  ProductID VARCHAR(50) NOT NULL,
  Name VARCHAR(50) NOT NULL,
  Weight NUMERIC(2) NOT NULL,
  Dimensions NUMERIC(2) NOT NULL,
  BinID VARCHAR(50) NOT NULL,
  CategoryID VARCHAR(50) NOT NULL,
  ManufacturerID INT NOT NULL,
  PRIMARY KEY (ProductID),
  FOREIGN KEY (BinID) REFERENCES BIN(BinID),
  FOREIGN KEY (CategoryID) REFERENCES PRODUCT_CATEGORY(CategoryID),
  FOREIGN KEY (ManufacturerID) REFERENCES MNAUFACTURER(ManufacturerID)
);

CREATE TABLE SHIPMENT
(
  ShipmentID VARCHAR(50) NOT NULL,
  ShipmentDate DATE NOT NULL,
  ProductID VARCHAR(50) NOT NULL,
  OrderID VARCHAR(50) NOT NULL,
  WarehouseID VARCHAR(50) NOT NULL,
  PRIMARY KEY (ShipmentID),
  FOREIGN KEY (ProductID) REFERENCES PRODUCT(ProductID),
  FOREIGN KEY (OrderID) REFERENCES ORDER(OrderID),
  FOREIGN KEY (WarehouseID) REFERENCES WAREHOUSE(WarehouseID)
);

CREATE TABLE PACKAGING
(
  PackagingID VARCHAR(50) NOT NULL,
  Status VARCHAR(50) NOT NULL,
  ShipmentID VARCHAR(50) NOT NULL,
  PRIMARY KEY (PackagingID),
  FOREIGN KEY (ShipmentID) REFERENCES SHIPMENT(ShipmentID)
);

CREATE TABLE WORKER
(
  WorkerID VARCHAR(50) NOT NULL,
  Name VARCHAR(50) NOT NULL,
  DOB DATE NOT NULL,
  Phone INT NOT NULL,
  Email VARCHAR(50) NOT NULL,
  Address VARCHAR(100) NOT NULL,
  WarehouseID VARCHAR(50) NOT NULL,
  PackagingID VARCHAR(50),
  PRIMARY KEY (WorkerID),
  FOREIGN KEY (WarehouseID) REFERENCES WAREHOUSE(WarehouseID),
  FOREIGN KEY (PackagingID) REFERENCES PACKAGING(PackagingID)
);

CREATE TABLE ORDERDETAIL
(
  OrderDetailID VARCHAR(50) NOT NULL,
  Quantity INT NOT NULL,
  UnitPrice NUMERIC(2) NOT NULL,
  OrderID VARCHAR(50) NOT NULL,
  PackagingID VARCHAR(50) NOT NULL,
  ProductID VARCHAR(50) NOT NULL,
  ClientID VARCHAR(50) NOT NULL,
  PRIMARY KEY (OrderDetailID),
  FOREIGN KEY (OrderID) REFERENCES ORDER(OrderID),
  FOREIGN KEY (PackagingID) REFERENCES PACKAGING(PackagingID),
  FOREIGN KEY (ProductID) REFERENCES PRODUCT(ProductID),
  FOREIGN KEY (ClientID) REFERENCES CLIENT(ClientID)
);

CREATE TABLE INVOICE
(
  InvoiceID VARCHAR(50) NOT NULL,
  Date DATE NOT NULL,
  OrderDetailID VARCHAR(50) NOT NULL,
  PRIMARY KEY (InvoiceID),
  FOREIGN KEY (OrderDetailID) REFERENCES ORDERDETAIL(OrderDetailID)
);
